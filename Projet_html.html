normalize1 <- function(table_letter, names_character){
  # Ajout des caractÃ¨res manquants Ã  la table d'effectif des caractÃ¨res d'un texte (occurrences nulles)
  
  # Inputs : 
  # "table_letter" : Tableau d'effectif des caractÃ¨res d'un texte
  # "names_character" : Ensemble des caractÃ¨res possibles
  
  # Output : 
  # "res" : Tableau complet d'effectif de chaque caractÃ¨re (dans un texte)
  
  N <- length(names_character)
  
  # Si aucun caractÃ¨re ne manque, on retourne le tableau
  if (length(table_letter)==N){ return(table_letter) }
  
  else{
    # CrÃ©ation d'un vecteur nul

    names(res) <- names_character
    
    # Remplace les occurrences nulles par les valeurs de table_letter lorsque le caractÃ¨re est prÃ©sent dans le texte
    for (character in names(table_letter))
    {

  }
}

naive_bayes <- function(index_test, index_train, matrix_logfreq)
{
  # PrÃ©diction de la langue des textes du jeu de donnÃ©es test Ã  partir du jeu d'apprentissage
    
  # Inputs :
  # "index_test" : vector, index des textes Ã  classifier
  # "index_train" : vector, index des textes du jeu d'apprentissage
  # "matrix_logfreq" : matrix, matrice des log-frÃ©quences de l'ensemble des textes
  
  # Output :
  # Liste des prÃ©dictions pour les textes du jeu de donnÃ©es test
  
  # Vecteur des prÃ©dictions
  y_pred <- c()
  
  # Estimation des paramÃ¨tres sur le jeu d'apprentissage
  param <- compute_param(index_train, matrix_logfreq)
  
  # Calcul des probabilitÃ©s marginales
  prob_en <- sum(matrix_logfreq[index_train, ncol(matrix_logfreq)]==1)/length(index_train)
  prob_fr <- 1-prob_en
  
  j <- 1
  
  # Boucle for sur le jeu de donnÃ©es test
  for(i in index_test)
  {
    # Calcul des probabilitÃ©s sans division du dÃ©nominateur
    pi_en <- prob_en * dmvnorm(matrix_logfreq[i,-c(ncol(matrix_logfreq))], param$mu_en, param$var_en)
    pi_fr <- prob_fr * dmvnorm(matrix_logfreq[i,-c(ncol(matrix_logfreq))], param$mu_fr, param$var_fr)
    
    y_pred[j] <- ifelse(pi_en > pi_fr, 1, -1)
    
    j <- j + 1
  }
  
  return(y_pred)
}

normalize2 <- function(bigram, character_list){
  # Ajout des caractÃ¨res manquants au bigram des caractÃ¨res d'un texte
  
  # Inputs : 
  # "bigram" : Bigram des caractÃ¨res d'un texte

  # Output : 
  # "res" : bigram complet de chaque caractÃ¨re (dans un texte)
  
  N = length(character_list)
  
  # Si aucune cellule ne manque, on retourne le tableau
  if ( (dim(bigram)[1]==N) && (dim(bigram)[2]==N) ) { return(bigram) }
  
  else{

  return (res)
  }
}

normalize3 <- function(bigram, epsilon=10e-7){
  # Ajout d'epsilon aux valeurs nulles des bigrams + suppression transition espace-espace
  
  # Inputs : 
  # "bigram" : bigram des caractÃ¨res d'une langue
  # "epsilon" : quantitÃ© ajoutÃ©e aux valeurs nulles
  
  # Output : 
  # "res" : bigram
  
  res = bigram
  
  for (letteri in rownames(res)){
    for (letterj in colnames(res)){
      # Remplace les occurrences nulles par epsilon
      if (res[letteri,letterj]==0){
        res[letteri,letterj] = epsilon
      }
    }
  }
  
  # On enlÃ¨ve les enchaÃ®nements espace-espace
  res[1,1] = 0
  
  return (res)
}

logprob_markov <- function(param, text_vector){
  # ProbabilitÃ© "ajustÃ©e" (On additionne log(1+f)) d'un texte modÃ©lisÃ© comme une chaine de markov de caractÃ¨re

  # Inputs :
  # "text_vector" : Vecteur de caractÃ¨re
  # "param" : ParamÃ¨tres de markov pi et A
  
  # Output : Langue estimÃ©e du texte (-1 ou 1)
  n = length(text_vector)
  res = log(1 + param$distrib_init[text_vector[1]])
  for (i in 2:n) {
    res = res + log(1 + param$mat_transition[text_vector[i-1],text_vector[i]])
  }
  return(res)
}

classifieur_Markovien <- function(text, Markov_FR, Markov_EN){
  # PrÃ©diction de la langue d'un texte en le modÃ©lisant comme une chaÃ®ne de Markov
  
  # Inputs :
  # "text" : Texte dont il faut prÃ©dire la langue (format : dataset$x[i])
  # "Markov_FR, Markov_EN" : ParamÃ¨tres de markov pi et A pour chaque langue
  
  # Output : PrÃ©diction de la langue du texte (-1 ou 1)
  
  # Transforme le texte en vecteur de caractÃ¨re
  text_vector = strsplit(as.vector(text), split="") %>% unlist()
  
  # Si le texte commence par un espace, on le supprime
  if (text_vector[1] == " "){text_vector = text_vector[-1]} 
  
  
  # Vraisemblance "ajustÃ©e" (on additionne log(1+f))
  ML_fr = logprob_markov(Markov_FR, text_vector)
  ML_en = logprob_markov(Markov_EN, text_vector)
  
  
  # Langue la plus probable selon les calculs
  if (ML_fr > ML_en){ return(-1) }
  else{ return(1) }
}

  Z = rep(0,T)

  Z[T] = which.max(logV[T,])
  for (t in (T-1):1) {
    Z[t] = S[t,Z[t+1]]
  }
